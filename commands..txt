1. venv
2. pip install django
3. django-admin startproject mysite . - створюємо новий проект
4. python manage.py migrate - створюємо DB файл
5. manage.py runserver - запуск сервера
6. python manage.py startapp blog - створює нові sub-app
7. add new app on INSTALLED_APPS from settings.py
8. create models in app/models.py
9. python manage.py createsuperuser
10. manage.py makemigrations blog - створює commit 
11. manage.py migrate blog
